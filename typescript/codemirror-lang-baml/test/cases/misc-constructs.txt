type UserId = string

==>

  IdentifierDecl,
  TypeExpr(ComplexTypeExpr(SimpleTypeExpr(IdentifierDecl)))
)))

)))

# Comments

// This is a comment
class User {
  name string // trailing comment
}

==>

Baml(
  TrailingComment,
  Decl(ClassDecl(
    IdentifierDecl,
    "{",
    ClassField(IdentifierDecl, TypeExpr(ComplexTypeExpr(SimpleTypeExpr(IdentifierDecl)))),
    TrailingComment,
    "}"
  ))
)
